git config --global user.name "Your Name"   输入配置的名字
git config --global user.email "email@example.com"  输入配置的邮箱
git init   初始化git仓库
git add  filename    把工作区的文件添加到暂存区
git status   查看版本库的状态
git commit -m "instructions"    把暂存区的文件添加到版本库
git diff    查看工作区中修改的内容
ssh-keygen -t rsa -C  “youremail@example.com”    在本地生成.ssh秘钥
